5.2 Compute distances

We compute within Malawi and Uganda separately to avoid a lengthy computation.

sources_dist <-
  map_dfr(
    unique(hh$country_pl),
    ~ {
        sf <- 
          hh %>% 
          dplyr::filter(country == "UGANDA") %>% 
          distinct(key) %>% 
          inner_join(hh_sf) %>% 
          st_as_sf()
        
        st_distance_df(
          sf, "key",
          sf, "key"
        ) %>%
        dplyr::filter(
          key.x != key.y,
          distance <= 3000
        )

    }
  )
