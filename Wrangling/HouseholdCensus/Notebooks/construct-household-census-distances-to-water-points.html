6.3 Create distance variables

  • dist_prim ~ distance to the primary water source they list in the survey
  • dist_wp ~ distance to the nearest water source
  • dist_ilc ~ distance to nearest ILC device
  • dist_dsw ~ distance to nearest DSW
  • dist_ilc_relative ~ dist_prim - dist_ilc
  • dist_dsw_relative ~ dist_prim - dist_dsw
  • dist_wp_relative ~ dist_prim - dist_wp

6.3.1 Distance to primary water source

If we can’t find a primary water source, we leave it as missing.

dist_prim <-
  hh %>% 
  transmute(
    household_id,
    wp_id = wp_id_pl
  ) %>% 
  inner_join(
    distances,
    by = c("wp_id", "household_id")
  )  %>% 
  transmute(
    household_id,
    dist_prim = distance
  )

6.3.2 Distance to nearest water point

dist_wp <-
  distances %>% 
  group_by(household_id) %>% 
  dplyr::filter(distance == min(distance, na.rm = T)) %>% 
  ungroup() %>% 
  distinct(household_id, distance, .keep_all = T) %>% 
  transmute(
    household_id,
    dist_wp = distance
  )

6.3.3 Distance to nearest ILC water point

dist_ilc <-
  wp %>% 
  dplyr::filter(wp_ilc == 1) %>% 
  transmute(
    wp_id
  ) %>% 
  inner_join(distances, by = "wp_id") %>% 
  group_by(household_id) %>% 
  dplyr::filter(distance == min(distance, na.rm = T)) %>% 
  ungroup() %>% 
  distinct(household_id, distance, .keep_all = T) %>% 
  transmute(
    household_id,
    dist_ilc = distance
  )

6.3.4 Distance to nearest DSW

dist_dsw <-
  wp %>% 
  dplyr::filter(wp_dsw == 1) %>% 
  transmute(
    wp_id
  ) %>% 
  inner_join(distances, by = "wp_id") %>% 
  group_by(household_id) %>% 
  dplyr::filter(distance == min(distance, na.rm = T)) %>% 
  ungroup() %>% 
  distinct(household_id, distance, .keep_all = T) %>% 
  transmute(
    household_id,
    dist_dsw = distance
  )

6.3.5 Combine distances

comb <-
  hh %>% 
  transmute(household_id) %>% 
  left_join(dist_prim) %>% 
  left_join(dist_wp) %>% 
  left_join(dist_ilc) %>% 
  left_join(dist_dsw) %>% 
  mutate(
    across(
      c(dist_wp, dist_ilc, dist_dsw),
      ~ ifelse(is.na(.x), 5000, .x)
    )
  )

6.3.6 Relative distance to nearest ILC and DSW to primary water source

comb <-
  comb %>% 
  mutate(
    across(
      c(dist_wp, dist_ilc, dist_dsw),
      ~ dist_prim - .x,
      .names = "{.col}_relative"
    )
  )
