5 Construct household distance

In this script, we compute the distances between households.

packages <-
  c(
    "tidyverse",
    "here",
    "haven",
    "skimr",
    "kableExtra",
    "sf"
  )

pacman::p_load(
  packages,
  character.only = TRUE
)

5.2 Compute distances

We separate data frame into district-level data frames to reduce run time and calculate distances between all water points in the same district.

hh_district <-
  hh_sf %>%
  group_by(district_id) %>%
  group_split %>%
  map(
    ~ st_distance_df(
      ., "key",
      ., "key"
    )
  ) %>%
  bind_rows() %>% 
  dplyr::filter(
    key.x != key.y
  )
