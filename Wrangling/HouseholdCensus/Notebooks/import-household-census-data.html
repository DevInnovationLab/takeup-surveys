2 Import household census data

packages <-
  c(
    "tidyverse",
    "here",
    "haven",
    "skimr",
    "kableExtra",
    "janitor"
  )

pacman::p_load(
  packages,
  character.only = TRUE
)

2.1 Load data

Corrections to the raw data were done by IPA. We load the dataset that already includes these corrections.

hh_raw_ug <- 
  read_dta(
    file.path(
      path_box,
      "Data - SHARED WITH IPA",
      "Uganda",
      "02_Baseline",
      "03_Data Cleaning",
      "3_post_hfc_data",
      "2_HH Census",
      "DIL Household Census_checked.dta"
    )
  )
hh_raw_ug_ilc <- 
  read_dta(
    file.path(
      path_box,
      "Data - SHARED WITH IPA",
      "Uganda",
      "02_Baseline",
      "03_Data Cleaning",
      "3_post_hfc_data",
      "2_HH Census",
      "ILC",
      "2_dta",
      "DIL Household Census_checked_V3.dta"
    )
  )
hh_raw_mw <-
  read_dta(
    file.path(
      path_box,
      "Data - SHARED WITH IPA",
      "Malawi",
      "02_Baseline",
      "03_Data Cleaning",
      "3_post_hfc_data",

      "DIL_Household_Census_checked_Malawi.dta"

2.2 Combine Uganda and Malawi data

We want to conduct the wrangling and HFCs for both countries at the same time, so we combine them here.

hhc_raw <-
  hh_raw_mw %>%
  bind_rows(hh_raw_ug) %>%
  bind_rows(hh_raw_ug_ilc)

2.3 Adjust variable formats

hhc_raw <-
  hhc_raw %>%
  as_factor(levels = "labels") %>%
  mutate(
    duration = as.numeric(duration),
    country_pl = ifelse(key == "uuid:447617f9-4a55-40a2-ac71-6a11f2802fff", "Malawi", country_pl)
  ) %>%
  remove_empty()
